# Interactive Examples

All demos run **entirely in the browser** — no backend required. Type a message and hit Enter (or click Send) to see real-time streaming in action.

## Svelte 5

The full `@traek/svelte` canvas with pan/zoom, branching, and mock streaming.

<SvelteDemo client:load />

**Try it:**

- Type a message and send it
- Click any node, then type another message to branch
- Pinch or scroll to zoom; drag the canvas to pan

**Source:**

```svelte
<script lang="ts">
	import { TraekCanvas, TraekEngine, type MessageNode } from '@traek/svelte';

	const engine = new TraekEngine();
	engine.addNode('Hello! How can I help?', 'assistant', { type: 'text' });

	async function handleSend(content: string, userNode: MessageNode) {
		const assistant = engine.addNode('', 'assistant', {
			type: 'text',
			parentIds: [userNode.id],
			status: 'streaming'
		});

		// Stream reply chunk by chunk
		const reply = `You said: "${content}". Stream your real AI response here.`;
		for (const char of reply) {
			await delay(20);
			engine.appendToNode(assistant.id, char);
		}
		engine.updateNode(assistant.id, { status: 'done' });
	}
</script>

<TraekCanvas {engine} onSendMessage={handleSend} />
```

## React 18

`@traek/react` with the same spatial canvas, powered by `@traek/core` state.

<ReactDemo client:load />

**Source:**

```tsx
import { TraekCanvas, useCreateTraekEngine } from '@traek/react';
import type { Node } from '@traek/core';

export default function App() {
	const engine = useCreateTraekEngine(() => {
		engine.addNode('Hello from React!', 'assistant', { type: 'text' });
	});

	async function handleSend(content: string, userNode: Node) {
		const assistant = engine.addNode('', 'assistant', {
			parentIds: [userNode.id],
			status: 'streaming'
		});

		const reply = `You said: "${content}".`;
		for (let i = 0; i < reply.length; i++) {
			await delay(18);
			engine.updateNode(assistant.id, { content: reply.slice(0, i + 1) });
		}
		engine.updateNode(assistant.id, { status: 'done' });
	}

	return <TraekCanvas engine={engine} onSendMessage={handleSend} />;
}
```

## Connect to a Real AI

Replace the mock streaming with a real backend call:

```ts
async function handleSend(content: string, userNode: MessageNode) {
	const assistant = engine.addNode('', 'assistant', {
		parentIds: [userNode.id],
		status: 'streaming'
	});

	const res = await fetch('/api/chat', {
		method: 'POST',
		headers: { 'Content-Type': 'application/json' },
		body: JSON.stringify({
			messages: engine.contextPath.map((n) => ({ role: n.role, content: n.content }))
		})
	});

	const reader = res.body!.getReader();
	const decoder = new TextDecoder();
	let done = false;
	while (!done) {
		const { value, done: d } = await reader.read();
		done = d;
		if (value) engine.appendToNode(assistant.id, decoder.decode(value));
	}
	engine.updateNode(assistant.id, { status: 'done' });
}
```
